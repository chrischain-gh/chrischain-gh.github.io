def complete_main_perf_fields(row):
    if row['Speed [kts]'] is np.nan and row['NMPG'] is np.nan \
    and row['Speed [mph]'] is not np.nan and row['GPH'] is not np.nan:
        row['Speed [kts]'] = 0.868976 * row['Speed [mph]']
        row['NMPG'] = 0.868976 * row['Speed [mph]'] / row['GPH']
    elif row['Speed [kts]'] is not np.nan and row['NMPG'] is np.nan \
    and row['GPH'] is not np.nan:
        row['NMPG'] = row['Speed [kts]'] / row['GPH']
    return row
